Calling REST endpoints from Power BI

July 25, 2025 
// Calculate the average number of days since the last scan for all projects
// The SUMMARIZE function builds an in-memory table with the __Days Since Latest Scan calculated for each project
// The AVERAGEX then calculates the average across all these values
Average Days Since Last Scan: 
    AVERAGEX(
        SUMMARIZE(ProjectMetrics, ProjectMetrics[Project Key], "METRIC VALUE", [__Days Since Latest Scan]),
        [METRIC VALUE]
    )

// Calculate the number of days since the last scan for a single project or return blank
__Days Since Latest Scan: 
    IF(HASONEVALUE(Projects[Project Key]), 
        CALCULATE(
            MIN(ProjectMetrics[Days since Scan]),
            FILTER(ProjectMetrics, ProjectMetrics[Scan DateTime]: MAX(ProjectMetrics[Scan DateTime]))
        ),
        BLANK()
    )

// Another example using number of issues
// We need to use the same technique here because we want to use the max date filter for each project (and each project may be different)
// If you just use MAX(Date) it only includes projects that were scanned on the latest date and ignores any older scanned projects
Total Issues: 
    SUMX(
        SUMMARIZE(ProjectMetrics, ProjectMetrics[Project Key], "METRIC VALUE", [__Total Issues]),
        [METRIC VALUE]
    )

__Total Issues: 
    VAR METRIC_NAME: "violations"
RETURN
    IF(HASONEVALUE(Projects[Project Key]), 
        CALCULATE(
            AVERAGEX(FILTER(ProjectMetrics, ProjectMetrics[Metric]: METRIC_NAME), [Value]),
            FILTER(ProjectMetrics, ProjectMetrics[Scan DateTime]: MAX(ProjectMetrics[Scan DateTime]))
        ),
        BLANK()
    )
